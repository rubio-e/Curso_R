### Incluso se puede crear un función para el cálculo

    CVAR <- function(x) {
      desviacion <- sd(x)
      media <- mean(x)
      result <- desviacion / media * (100)
      return(result)
    }

    CVAR(DN)

    ## [1] 45.68016
